Código para criação de Banco de Dados (MER):

Table Aluno {
  id int [pk]
  nome varchar
  turma varchar
}

Table Ferramenta {
  id_ferramenta int [pk]
  nome_ferramenta varchar
  descricao text
}

Table Emprestimo {
  id_emprestimo int [pk]
  id_aluno int [ref: > Aluno.id]
  id_ferramenta int [ref: > Ferramenta.id_ferramenta]
  data_retirada date
  data_devolucao date
}
